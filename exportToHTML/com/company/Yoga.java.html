package com.company;

import java.util.ArrayList;
import java.util.Scanner;

public class Yoga {



    public static void yoga() {
        String response;
        String stop = "Stopp";

        ArrayList<String> yogabooking = new ArrayList<>();
        ArrayList<String> yogaavailable = new ArrayList<>(8);
        yogaavailable.add("1a");
        yogaavailable.add("2a");
        yogaavailable.add("3a");
        yogaavailable.add("1b");
        yogaavailable.add("2b");
        yogaavailable.add("3b");
        yogaavailable.add("1c");
        yogaavailable.add("2c");
        yogaavailable.add("3c");



        if (yogabooking.size() < 8 )
            do {
                System.out.println("Välj plats eller skriv stopp för att avsluta:");
                for (int counter = 0; counter < yogaavailable.size(); counter++) {
                    System.out.println(yogaavailable.get(counter));
                }
                Scanner input = new Scanner(System.in);
                response = input.nextLine();

                if (!yogaavailable.contains(response) && yogabooking.contains(response) && !response.equalsIgnoreCase(stop)) {
                    System.out.println("Tyvärr, platsen är redan upptagen.");

                }
                else if (!yogaavailable.contains(response) && !yogabooking.contains(response) && !response.equalsIgnoreCase(stop)) {
                    System.out.println("Ange en giltig plats.");
                }

                else {
                    yogaavailable.remove(response);
                    yogabooking.add(response);
                    System.out.printf("Klart!%n");
                }
            }
            while (!response.equalsIgnoreCase(stop));


    }

}
